class User {
  - userId: String
  - username: String
  - email: String
  - password: String
  + register()
  + login()
  + updateProfile()
}

class LoggedUser {
  - preferences: List<String>
  - calendarSynced: boolean
  + syncCalendar()
  + joinEvent()
  + voteActivity()
  + initiateGroupPlanning()
}

class Admin {
  + monitorSystem()
  + manageEvents()
}

class SocialGroup {
  - groupId: String
  - name: String
  - commonPreferences: String
  + addMember()
  + removeMember()
  + startVoting()
}

class Event {
  - eventId: String
  - title: String
  - location: String
  - time: String
  - price: double
  - vibe: String
  + createEvent()
  + updateEvent()
  + calculateCompatibility()
}

class Vote {
  - voteId: String
  - decision: boolean
  + submitVote()
}

class ChatMessage {
  - messageId: String
  - content: String
  - timestamp: String
  + sendMessage()
}

class CalendarSyncService {
  + syncGoogleCalendar()
  + findCommonFreeTime()
}

class AIRecommendationEngine {
  + generateSuggestions()
  + refineFilters()
  + suggestNextStep()
}

class PaymentService {
  + processPayment()
  + confirmPayment()
}

class LocationOptimizer {
  + computeBestMeetingPoint()
}
